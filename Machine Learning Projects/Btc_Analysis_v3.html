import re
def remove_emojis(data):
    emoj = re.compile("["
        u"\U0001F600-\U0001F64F"  # emoticons
        u"\U0001F300-\U0001F5FF"  # symbols & pictographs
        u"\U0001F680-\U0001F6FF"  # transport & map symbols
        u"\U0001F1E0-\U0001F1FF"  # flags (iOS)
        u"\U00002500-\U00002BEF"  # chinese char
        u"\U00002702-\U000027B0"
        u"\U00002702-\U000027B0"
        u"\U000024C2-\U0001F251"
        u"\U0001f926-\U0001f937"
        u"\U00010000-\U0010ffff"
        u"\u2640-\u2642" 
        u"\u2600-\u2B55"
        u"\u200d"
        u"\u23cf"
        u"\u23e9"
        u"\u231a"
        u"\ufe0f"  # dingbats
        u"\u3030"
                      "]+", re.UNICODE)
    return re.sub(emoj, '', data)

# Funcin al DataFrame
def limpiar_tweets(df):
  
  # Eliminamos links
  df['text'] = df['text'].str.replace(r'http\S+', '')
  # Eliminamos Emojis
  df['text'] = df['text'].apply(lambda x: remove_emojis(str(x)))
  # Quedarnos solo con las letras
  df['text'] = df['text'].str.replace(r'[^a-zA-Z0-9]', ' ')
  # Eliminar dos espacios o mas seguidos.
  df['text'] = df['text'].str.replace('[\s]+', ' ',)
  # Pasar a minusculas
  df['text'] = df['text'].apply(lambda x: x.lower())

  return df

Una vez se ha conseguido la limpieza de los twits, se procede a quitar las stopwords en ingls. Las stopwords son palabras que no aportan nada al analisis, como por ejemplo las preposiciones. Al eliminar estas palabras, eliminamos la informacin de bajo nivel de nuestro texto para centrarnos ms en la informacin importante. La eliminacin de estas palabras no muestra ninguna consecuencia negativa en el modelo que entrenamos para nuestra tarea, adems de reducir el tiempo de entrenamiento debido al menor nmero de tokens implicados.

Una vez se han aplicado estas transformaciones, el data frame resultante ya puede ser utilizado para el analisis.

def remove_stopwords(text, is_lower_case=False):
    tokens = tokenizer.tokenize(text)
    tokens = [token.strip() for token in tokens]
    if is_lower_case:
        filtered_tokens = [token for token in tokens if token not in stopword_list]
    else:
        filtered_tokens = [token for token in tokens if token.lower() not in stopword_list]
    filtered_text = ' '.join(filtered_tokens)    
    return filtered_text

from vaderSentiment.vaderSentiment import SentimentIntensityAnalyzer

Para hacer el analisis, se utiliza el paquete Vader, el cual utiliza una lista de caractersticas lxicas (palabras) que se etiquetan como positivas o negativas segn su orientacin semntica para calcular el sentimiento del texto. Vader sentiment devuelve la probabilidad de que una frase de entrada sea positiva, negativa o neutra.

Para ser considerado un twit positivo, tiene que tener al menos un valor de 5% positivo, ya que si el umbral es mayor, la gran mayora de twits estaran clasificados como neutrales.

def sentiment_scores(df):
    # Create a SentimentIntensityAnalyzer object.
    sid_obj = SentimentIntensityAnalyzer()
    # polarity_scores method of SentimentIntensityAnalyzer
    # oject gives a sentiment dictionary.
    # which contains pos, neg, neu, and compound scores.
    df['sentiment_dict'] = df['cleaned_text'].apply(lambda x: sid_obj.polarity_scores(x))  
    df['pcte_negativo'] = df['sentiment_dict'].apply(lambda x: x['neg'])
    df['pcte_neutral'] = df['sentiment_dict'].apply(lambda x: x['neu'])
    df['pcte_postivo'] = df['sentiment_dict'].apply(lambda x: x['pos'])
    # df['pcte_neutral'] = df['sentiment_dict']['neu']
    # df['pcte_positivo'] = df['sentiment_dict']['pos']

    # sentiment_dict = sid_obj.polarity_scores(sentence)
    # print("Overall sentiment dictionary is : ", sentiment_dict)
    # print("sentence was rated as ", sentiment_dict['neg']*100, "% Negative")
    # print("sentence was rated as ", sentiment_dict['neu']*100, "% Neutral")
    # print("sentence was rated as ", sentiment_dict['pos']*100, "% Positive")
    # print("Sentence Overall Rated As", end = " ")
    # decide sentiment as positive, negative and neutral
    
    df['target'] = np.where((df['sentiment_dict'].apply(lambda x: x['compound'])) >= 0.05, 'Positivo', 
                            np.where((df['sentiment_dict'].apply(lambda x: x['compound'])) <= - 0.05, 'Negativo', 'Neutral'))
    


    return df

import numpy as np

import seaborn as sns

def make_confusion_matrix(cf,
                          group_names=None,
                          categories='auto',
                          count=True,
                          percent=True,
                          cbar=True,
                          xyticks=True,
                          xyplotlabels=True,
                          sum_stats=True,
                          figsize=None,
                          cmap='Blues',
                          title='Confusion matrix for tweets classification'):
    '''
    This function will make a pretty plot of an sklearn Confusion Matrix cm using a Seaborn heatmap visualization.
    Arguments
    ---------
    cf:            confusion matrix to be passed in
    group_names:   List of strings that represent the labels row by row to be shown in each square.
    categories:    List of strings containing the categories to be displayed on the x,y axis. Default is 'auto'
    count:         If True, show the raw number in the confusion matrix. Default is True.
    normalize:     If True, show the proportions for each category. Default is True.
    cbar:          If True, show the color bar. The cbar values are based off the values in the confusion matrix.
                   Default is True.
    xyticks:       If True, show x and y ticks. Default is True.
    xyplotlabels:  If True, show 'True Label' and 'Predicted Label' on the figure. Default is True.
    sum_stats:     If True, display summary statistics below the figure. Default is True.
    figsize:       Tuple representing the figure size. Default will be the matplotlib rcParams value.
    cmap:          Colormap of the values displayed from matplotlib.pyplot.cm. Default is 'Blues'
                   See http://matplotlib.org/examples/color/colormaps_reference.html
                   
    title:         Title for the heatmap. Default is None.
    '''


    # CODE TO GENERATE TEXT INSIDE EACH SQUARE
    blanks = ['' for i in range(cf.size)]

    if group_names and len(group_names)==cf.size:
        group_labels = ["{}\n".format(value) for value in group_names]
    else:
        group_labels = blanks

    if count:
        group_counts = ["{0:0.0f}\n".format(value) for value in cf.flatten()]
    else:
        group_counts = blanks

    if percent:
        group_percentages = ["{0:.2%}".format(value) for value in cf.flatten()/np.sum(cf)]
    else:
        group_percentages = blanks

    box_labels = [f"{v1}{v2}{v3}".strip() for v1, v2, v3 in zip(group_labels,group_counts,group_percentages)]
    box_labels = np.asarray(box_labels).reshape(cf.shape[0],cf.shape[1])


    # CODE TO GENERATE SUMMARY STATISTICS & TEXT FOR SUMMARY STATS
    if sum_stats:
        #Accuracy is sum of diagonal divided by total observations
        accuracy  = np.trace(cf) / float(np.sum(cf))

        #if it is a binary confusion matrix, show some more stats
        if len(cf)==2:
            #Metrics for Binary Confusion Matrices
            precision = cf[1,1] / sum(cf[:,1])
            recall    = cf[1,1] / sum(cf[1,:])
            f1_score  = 2*precision*recall / (precision + recall)
            stats_text = "\n\nAccuracy={:0.3f}\nPrecision={:0.3f}\nRecall={:0.3f}\nF1 Score={:0.3f}".format(
                accuracy,precision,recall,f1_score)
        else:
            stats_text = "\n\nAccuracy={:0.3f}".format(accuracy)
    else:
        stats_text = ""


    # SET FIGURE PARAMETERS ACCORDING TO OTHER ARGUMENTS
    if figsize==None:
        #Get default figure size if not set
        figsize = plt.rcParams.get('figure.figsize')

    if xyticks==False:
        #Do not show categories if xyticks is False
        categories=False


    # MAKE THE HEATMAP VISUALIZATION
    plt.figure(figsize=figsize)
    sns.heatmap(cf,annot=box_labels,fmt="",cmap=cmap,cbar=cbar,xticklabels=categories,yticklabels=categories)

    if xyplotlabels:
        plt.ylabel('True label')
        plt.xlabel('Predicted label' + stats_text)
    else:
        plt.xlabel(stats_text)
    
    if title:
        plt.title(title)

Esto se debe a que en lugar de calcular el porcentaje de elementos en los que coinciden los evaluadores, el coeficiente de Cohen tiene en cuenta de que se puede acertar la clase aleatoriamente, por lo tanto siempre va a ser menor que la precisin. An asi, este valor sigue siendo muy positivo, indicando que nuestro modelo esta funcionando como se esperaba.
